==========================================================
ENTERPRISE BGP EXPERT CONFIGURATION TEMPLATE
Includes: eBGP, BFD, Prefix-Filtering, and Route-Maps
==========================================================

1. IP ADDRESS & CONNECTIVITY SETUP
----------------------------------
conf t
 interface Loopback0
  description *** BGP SOURCE INTERFACE ***
  ip address 1.1.1.1 255.255.255.255
 !
 interface GigabitEthernet0/0/0
  description *** LINK TO ISP / EXTERNAL PEER ***
  ip address 10.255.255.1 255.255.255.252
  bfd interval 300 min_rx 300 multiplier 3

2. PREFIX-LISTS & ROUTE-MAPS (Security & Policy)
------------------------------------------------
 ! Define what routes we will ADVERTISE to the ISP
 ip prefix-list PL_OUT_TO_ISP seq 5 permit 192.168.10.0/24
 ip prefix-list PL_OUT_TO_ISP seq 10 permit 192.168.20.0/24
 
 ! Define what routes we will ACCEPT from the ISP
 ip prefix-list PL_IN_FROM_ISP seq 5 permit 0.0.0.0/0
 
 ! Create Route-Maps to apply the Prefix-Lists
 route-map RM_ISP_OUT permit 10
  match ip address prefix-list PL_OUT_TO_ISP
 !
 route-map RM_ISP_IN permit 10
  match ip address prefix-list PL_IN_FROM_ISP

3. BGP ROUTING PROCESS CONFIGURATION
------------------------------------
router bgp 65001
 b_gp log-neighbor-changes
 
 # Neighbor Definition
 neighbor 10.255.255.2 remote-as 65100
 neighbor 10.255.255.2 description *** PEER_TO_ISP ***
 neighbor 10.255.255.2 update-source Loopback0
 neighbor 10.255.255.2 fall-over bfd  <-- (Enables sub-second link failure detection)
 
 # Address Family IPv4 (Modern Configuration Style)
 address-family ipv4
  neighbor 10.255.255.2 activate
  neighbor 10.255.255.2 send-community  <-- (Sends BGP tags for policy)
  neighbor 10.255.255.2 route-map RM_ISP_IN in
  neighbor 10.255.255.2 route-map RM_ISP_OUT out
  
  # Injecting Local Networks into BGP
  network 192.168.10.0 mask 255.255.255.0
  network 192.168.20.0 mask 255.255.255.0
  
  # Auto-summary must be disabled
  no auto-summary
 exit-address-family

4. BGP EXPERT VERIFICATION COMMANDS
------------------------------------
# Check overall BGP Status and Neighbors:
show ip bgp summary

# View specific routes learned from the ISP:
show ip bgp neighbors 10.255.255.2 received-routes

# View routes being advertised to the ISP:
show ip bgp neighbors 10.255.255.2 advertised-routes

# Check BFD Session (Link health):
show bfd neighbors
