  bgp graceful-restart restart-time 120
  bgp graceful-restart stalepath-time 360
  
  ! Neighbor Configuration for WAN Links
  neighbor 10.160.1.2 remote-as 65002
  neighbor 10.160.1.2 description BGP_ISP_Link_2
  neighbor 10.160.1.2 password BGP_Pass_123
  neighbor 10.160.1.2 timers 3 9
  neighbor 10.160.1.2 timers connect 10
  
  ! Address Family Configuration

    neighbor 10.160.1.2 activate
    neighbor 10.160.1.2 soft-reconfiguration inbound
    neighbor 10.160.1.2 route-map EXPORT-ROUTES out
    network 10.0.0.0 mask 255.0.0.0
    network 10.10.0.0 mask 255.255.255.0
    network 10.20.0.0 mask 255.255.255.0
  exit-address-family
